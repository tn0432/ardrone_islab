 9function R=rotsym(a,b)
10if (a=='x')
11    c=cos(b);
12    s=sin(b);
13    R=[1  0  0;
14       0  c -s;
15       0  s  c];
16end;
17if (a=='y')
18    c=cos(b);
19    s=sin(b);
20    R=[ c  0  s;
21        0  1  0;
22       -s  0  c];
23end;
24if (a=='z')
25    c=cos(b);
26    s=sin(b);
27    R=[c -s  0;
28       s  c  0;
29       0  0  1];
30end;
31end
